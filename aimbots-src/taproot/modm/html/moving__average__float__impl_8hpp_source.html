67void
68modm::filter::MovingAverage<double, N>::update(const double& input){
69 buffer[index] = input;
70
71 index++;
72 if (index >= N) {
73 index = 0;
74 }
75
76 sum = 0;
77 for (Index i = 0; i < N; ++i) {
78 sum+= buffer[i];
79 }
