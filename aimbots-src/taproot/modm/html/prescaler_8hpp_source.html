65 template<class Iterator>
66 static constexpr Result
67 from_iterator(T input_frequency,
68 T desired_frequency,
69 Iterator prescalers)
70 {
71 const double desired = double(input_frequency) / desired_frequency;
72 uint32_t prescaler_floor{*prescalers.begin()};
73 uint32_t prescaler_ceiling{*prescalers.begin()};
74 uint32_t ii_floor{0}, ii_ceiling{0};
75 if (desired <= prescaler_floor or prescalers.size() == 1)
76 return {input_frequency, desired_frequency, 0, prescaler_floor};
77 for (uint32_t prescaler : prescalers)
78 {
79 prescaler_ceiling = prescaler; ii_ceiling++;
80 if (prescaler_floor <= desired and desired < prescaler_ceiling)
81 break;
82 prescaler_floor = prescaler; ii_floor++;
83 }
84 const T baud_lower = input_frequency / prescaler_ceiling;
85 const T baud_upper = input_frequency / prescaler_floor;
86 const double baud_middle = (baud_upper + baud_lower) / 2.0;
87
88 if (desired_frequency <= baud_middle)
89 return {input_frequency, desired_frequency, ii_ceiling-1, prescaler_ceiling};
90 return {input_frequency, desired_frequency, ii_floor-1, prescaler_floor};
91 }

111 struct linear_range_iterator
112 {
113 uint32_t state;
114 const Function modifier;
115 constexpr uint32_t operator*() const { return modifier(state); }
116 constexpr linear_range_iterator& operator++() { state++; return *this; }
117 constexpr bool operator!=(const linear_range_iterator &o) const { return state != o.state; }
118 };
119 struct linear_range
120 {
121 const uint32_t m_begin;
122 const uint32_t m_end;
123 const Function modifier;
124 constexpr linear_range_iterator begin() const { return linear_range_iterator{m_begin, modifier}; }
125 constexpr linear_range_iterator end() const { return linear_range_iterator{m_end, modifier}; }
126 constexpr size_t size() const { return m_end - m_begin; }
127 };
128 linear_range range{begin, end+1, prescaler_modifier};
129 return from_iterator(input_frequency, desired_frequency, range);

134 static constexpr Result
135 from_range(T input_frequency,
136 T desired_frequency,
137 uint32_t begin, uint32_t end)
138 {
139 const double desired = double(input_frequency) / desired_frequency;
140 const uint32_t prescaler_floor = std::max(uint32_t(std::floor(desired)), begin);
141 const uint32_t prescaler_ceiling = std::min(uint32_t(std::ceil(desired)), end);
142 const T baud_lower = input_frequency / prescaler_ceiling;
143 const T baud_upper = input_frequency / prescaler_floor;
144 const double baud_middle = (baud_upper + baud_lower) / 2.0;
145 if (desired_frequency <= baud_middle)
146 return {input_frequency, desired_frequency, prescaler_ceiling - begin, prescaler_ceiling};
147 return {input_frequency, desired_frequency, prescaler_floor - begin, prescaler_floor};
148 }

154 static constexpr Result
155 from_power(T input_frequency,
156 T desired_frequency,
157 uint32_t begin, uint32_t end)
158 {
159 const double desired = double(input_frequency) / desired_frequency;
160 uint32_t ii{0}; uint32_t power{begin};
161 for (; power < end; power *= 2, ii++)
162 if (desired <= double(power * 4.0/3)) break;
163 return {input_frequency, desired_frequency, ii, power};
164 }
