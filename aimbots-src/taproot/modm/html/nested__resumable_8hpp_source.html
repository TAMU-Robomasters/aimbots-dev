29template< uint8_t Levels = 1>
30class NestedResumable
31{
32 static_assert(Levels > 0, "The number of nesting levels must be at least 1!");
33
34protected:
36 NestedResumable()
37 : rfLevel(0)
38 {
39 for (rf::State &level : rfStateArray)
40 {
41 level = rf::Stopped;
42 }
43 }
44
45public:
47 inline void
48 stopResumable()
49 {
50 uint_fast8_t level = rfLevel;
51 while (level < Levels)
52 {
53 rfStateArray[level++] = rf::Stopped;
54 }
55 }
56
58 bool inline
59 isResumableRunning() const
60 {
61 return !isStoppedRf();
62 }
63
65 int8_t inline
66 getResumableDepth() const
67 {
68 return static_cast<int8_t>(rfLevel) - 1;
69 }
70
71#ifdef __DOXYGEN__
79 modm::ResumableResult< ReturnType >
80 resumable function(...);
81#endif
82
83protected:
85
88 rf::State inline
89 pushRf(uint8_t /*index*/)
90 {
91 return rfStateArray[rfLevel++];
92 }
93
96 void inline
97 popRf()
98 {
99 rfLevel--;
100 }
101
102 // invalidates the parent nesting level
103 // @warning be aware in which nesting level you call this! (before popRf()!)
104 void inline
105 stopRf(uint8_t /*index*/)
106 {
107 rfStateArray[rfLevel-1] = rf::Stopped;
108 }
109
112 void inline
113 setRf(rf::State state, uint8_t /*index*/)
114 {
115 rfStateArray[rfLevel-1] = state;
116 }
117
120 bool inline
121 nestingOkRf() const
122 {
123 return modm_assert_continue_fail_debug(rfLevel < Levels, "rf.nest",
124 "Called too many nested ResumableFunctions of the same class!", this);
125 }
126
128 bool inline
129 isStoppedRf() const
130 {
131 return (rfStateArray[rfLevel] == rf::Stopped);
132 }
133
135 template<uint8_t index>
136 static void
137 checkRfFunctions()
138 {}
139
141 template<bool isNested>
142 static void
143 checkRfType()
144 {
145 static_assert(isNested == true, "You cannot declare an index for a _nested_ resumable function!");
146 }
148private:
149 uint_fast8_t rfLevel;
150 rf::State rfStateArray[Levels];
151};
