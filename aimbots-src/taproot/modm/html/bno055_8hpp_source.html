591template < class I2cMaster >
592class Bno055 : public bno055, public modm::I2cDevice<I2cMaster, 4>
593{
594public:
596 inline Bno055(Data &data, uint8_t address=addr()):
597 I2cDevice<I2cMaster,4>(address), data(data) {}
598
599 inline modm::ResumableResult<bool>
600 configure(OperationMode mode=OperationMode::NDOF)
601 {
602 return updateRegister(Register::OPR_MODE, mode);
603 }
604
605 inline modm::ResumableResult<bool>
606 readData()
607 {
608 return readRegister(Register::ACCEL_DATA_X_LSB, (uint8_t*)&data.r, Data::size);
609 }
610
611 inline modm::ResumableResult<bool>
612 enableExternalClock()
613 {
614 return updateRegister(Register::SYS_TRIGGER, SystemTrigger::CLK_SEL, Registers_t(0));
615 }
616
617public:
618 inline modm::ResumableResult<bool>
619 updateRegister(Register reg, Registers_t setMask, Registers_t clearMask = Registers_t(0xff))
620 {
621 RF_BEGIN();
622 RF_CALL(setPageId(reg));
623
624 buffer[0] = uint8_t(reg);
625 this->transaction.configureWriteRead(buffer, 1, buffer, 1);
626 RF_CALL( this->runTransaction() );
627
628 buffer[1] = (buffer[0] & ~clearMask.value) | setMask.value;
629 buffer[0] = uint8_t(reg);
630 this->transaction.configureWrite(buffer, 2);
631 RF_END_RETURN_CALL( this->runTransaction() );
632 }
633
634 inline modm::ResumableResult<bool>
635 readRegister(Register reg, uint8_t *output, size_t length=1)
636 {
637 RF_BEGIN();
638 RF_CALL(setPageId(reg));
639
640 buffer[0] = uint8_t(reg);
641 this->transaction.configureWriteRead(buffer, 1, output, length);
642 RF_END_RETURN_CALL( this->runTransaction() );
643 }
644
645public:
646 inline Data&
647 getData()
648 { return data; }
649
650protected:
651 inline modm::ResumableResult<bool>
652 setPageId(Register regi)
653 {
654 const uint8_t reg = uint8_t(regi);
655 RF_BEGIN();
656
657 if ((reg ^ prev_reg) & 0x80) {
658 buffer[0] = uint8_t(Register::PAGE_ID);
659 buffer[1] = reg >> 7;
660 this->transaction.configureWrite(buffer, 2);
661 buffer[2] = RF_CALL( this->runTransaction() );
662 if (buffer[2]) prev_reg = reg;
663 RF_RETURN(static_cast<bool>(buffer[2]));
664 }
665
666 RF_END_RETURN(true);
667 }
668
669private:
670 Data &data;
671 uint8_t buffer[3];
672 uint8_t prev_reg;
673};
