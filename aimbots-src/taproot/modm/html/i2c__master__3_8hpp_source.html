46 {
47 using Connector = GpioConnector<Peripheral::I2c3, Signals...>;
48 using Scl = typename Connector::template GetSignal<Gpio::Signal::Scl>;
49 using Sda = typename Connector::template GetSignal<Gpio::Signal::Sda>;
50 static_assert(sizeof...(Signals) == 2 and
51 Connector::template IsValid<Scl> and Connector::template IsValid<Sda>,
52 "I2cMaster3::connect() requires one Scl and one Sda signal!");

54 (pullups == PullUps::Internal) ? Gpio::InputType::PullUp : Gpio::InputType::Floating;

56 Connector::disconnect();
57 Scl::configure(input);
58 Sda::configure(input);
59 Scl::setOutput(Gpio::OutputType::OpenDrain);
60 Sda::setOutput(Gpio::OutputType::OpenDrain);
61 if (reset != ResetDevices::NoReset) resetDevices<Scl, uint32_t(reset)>();
62 Connector::connect();
63 }

71 template<class SystemClock, baudrate_t baudrate=kBd(100), percent_t tolerance=pct(5)>
72 static modm_always_inline void
73 initialize()
74 {
75 // calculate the expected clock ratio
76 constexpr uint8_t scalar = (baudrate <= 100'000) ? 2 : ((baudrate <= 300'000) ? 3 : 25);
77 constexpr uint16_t range_begin = (scalar == 2) ? 4 : 1;
78
79 constexpr auto result = Prescaler::from_range(
80 SystemClock::I2c3 / scalar, baudrate, range_begin, 4095);
81 assertBaudrateInTolerance< result.frequency, baudrate, tolerance >();
82
83 // the final prescaler value is augmented with the F/S and DUTY bit.
84 constexpr uint32_t prescaler = result.prescaler |
85 ((scalar >= 3) ? (1 << 15) : 0) |
86 ((scalar == 25) ? (1 << 14) : 0);
87
88 // peripheral frequency clock
89 constexpr uint8_t freq = SystemClock::I2c3 / 1'000'000;
90
91 // maximum rise time: assuming its linear:
92 // 1000ns @ 100kHz and 300ns @ 400kHz
93 // => y = x * m + b, with m = -2.3333ns/kHz, b = 1'233.3333ns
94 constexpr float max_rise_time = -2.333333f * (float(baudrate) / 1'000.f) + 1'233.333333f;
95 // calculate trise
96 constexpr float trise_raw = max_rise_time < 0 ? 0 : std::floor(max_rise_time / (1'000.f / freq));
97 constexpr uint8_t trise = trise_raw > 62 ? 63 : (trise_raw + 1);
98
99 initializeWithPrescaler(freq, trise, prescaler);
100 }
