65template<typename T, unsigned int ScaleFactor>
66void
67modm::Pid<T, ScaleFactor>::update(const T& input, bool externalLimitation)
68{
69 bool limitation = externalLimitation;
70
71 T tempErrorSum = errorSum + input;
72 if (tempErrorSum > this->parameter.maxErrorSum) {
73 tempErrorSum = this->parameter.maxErrorSum;
74 }
75 else if (tempErrorSum < -this->parameter.maxErrorSum) {
76 tempErrorSum = -this->parameter.maxErrorSum;
77 }
78
79 WideType tmp = 0;
80 tmp += static_cast<WideType>(this->parameter.kp) * input;
81 tmp += static_cast<WideType>(this->parameter.ki) * (tempErrorSum);
82 tmp += static_cast<WideType>(this->parameter.kd) * (input - this->lastError);
83
84 tmp = tmp / ScaleFactor;
85
86 if (tmp > this->parameter.maxOutput) {
87 this->output = this->parameter.maxOutput;
88 limitation = true;
89 }
90 else if (tmp < -this->parameter.maxOutput) {
91 this->output = -this->parameter.maxOutput;
92 limitation = true;
93 }
94 else {
95 this->output = tmp;
96 }
97
98 // If an external limitation (saturation somewhere in the control loop) is
99 // applied the error sum will only be decremented, never incremented.
100 // This is done to help the system to leave the saturated state.
101 if (not limitation or (std::abs(tempErrorSum) < std::abs(this->errorSum)))
102 {
103 this->errorSum = tempErrorSum;
104 }
105
106 this->lastError = input;
107}
