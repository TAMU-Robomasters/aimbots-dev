21template< class SystemClock, modm::frequency_t frequency, modm::percent_t tolerance >
22void
23modm::platform::Adc3::initialize()
24{
25 constexpr auto result = modm::Prescaler::from_list(SystemClock::Adc, frequency, {2,4,6,8});
26 static_assert(result.frequency <= 36000000, "Generated ADC frequency is above maximum frequency!");
27 assertBaudrateInTolerance<result.frequency, frequency, tolerance >();
28
29 Rcc::enable<Peripheral::Adc3>();
30 ADC3->CR2 |= ADC_CR2_ADON; // switch on ADC
31
32 setPrescaler(Prescaler{result.index});
33}

64bool
65modm::platform::Adc3::setChannel(const Channel channel,
66 const SampleTime sampleTime)
67{
68 if (uint32_t(channel) > 18) return false;
69 // clear number of conversions in the sequence
70 // and set number of conversions to 1
71 ADC3->SQR1 = 0;
72 ADC3->SQR2 = 0;
73 ADC3->SQR3 = uint32_t(channel) & 0x1f;
74
75 setSampleTime(channel, sampleTime);
76 return true;
77}

85bool
86modm::platform::Adc3::addChannel(const Channel channel,
87 const SampleTime sampleTime)
88{
89 // read channel count
90 uint8_t channel_count = (ADC3->SQR1 & ADC_SQR1_L) >> 20;
91 ++channel_count;
92 if(channel_count > 0x0f) return false; // emergency exit
93 // write channel number
94 if(channel_count < 6) {
95 ADC3->SQR3 |=
96 (uint32_t(channel) & 0x1f) << (channel_count*5);
97 } else if(channel_count < 12) {
98 ADC3->SQR2 |=
99 (uint32_t(channel) & 0x1f) << ((channel_count-6)*5);
100 } else {
101 ADC3->SQR1 |=
102 (uint32_t(channel) & 0x1f) << ((channel_count-12)*5);
103 }
104 // update channel count
105 ADC3->SQR1 = (ADC3->SQR1 & ~ADC_SQR1_L) | (channel_count << 20);
106
107 setSampleTime(channel, sampleTime);
108 return true;
109}

111void
112modm::platform::Adc3::setSampleTime(const Channel channel,
113 const SampleTime sampleTime)
114{
115 if (uint32_t(channel) < 10) {
116 ADC3->SMPR2 |= uint32_t(sampleTime)
117 << (uint32_t(channel) * 3);
118 }
119 else {
120 ADC3->SMPR1 |= uint32_t(sampleTime)
121 << ((uint32_t(channel)-10) * 3);
122 }
123}

165uint16_t
166modm::platform::Adc3::readChannel(Channel channel)
167{
168 if (!setChannel(channel)) return 0;
169
170 startConversion();
171 // wait until the conversion is finished
172 while (!isConversionFinished())
173 ;
174
175 return getValue();
176}
