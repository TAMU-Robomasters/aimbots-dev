49 template< class SystemClock, percent_t tolerance=pct(0) >
50 static void
51 initialize()
52 {
53 static_assert(SystemClock::Frequency < (1ull << 24)*8*4,
54 "HLCK is too fast for the SysTick to run at 4Hz!");
55 if constexpr (SystemClock::Frequency < 8'000'000)
56 {
57 constexpr auto result = Prescaler::from_range(
58 SystemClock::Frequency, 4, 1, (1ul << 24)-1);
59 PeripheralDriver::assertBaudrateInTolerance< result.frequency, 4, tolerance >();
60
61 us_per_Ncycles = ((1ull << Ncycles) * 1'000'000ull) / SystemClock::Frequency;
62 ms_per_Ncycles = ((1ull << Ncycles) * 1'000ull) / SystemClock::Frequency;
63 enable(result.index, false);
64 }
65 else
66 {
67 constexpr auto result = Prescaler::from_range(
68 SystemClock::Frequency/8, 4, 1, (1ul << 24)-1);
69 PeripheralDriver::assertBaudrateInTolerance< result.frequency, 4, tolerance >();
70
71 us_per_Ncycles = ((1ull << Ncycles) * 8'000'000ull) / SystemClock::Frequency;
72 ms_per_Ncycles = ((1ull << Ncycles) * 8'000ull) / SystemClock::Frequency;
73 enable(result.index, true);
74 }
75 }
