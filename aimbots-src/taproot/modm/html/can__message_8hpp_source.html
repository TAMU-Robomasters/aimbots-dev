26struct Message
27{
28 inline Message(uint32_t inIdentifier = 0, uint8_t inLength = 0) :
29 identifier(inIdentifier), flags(), length(inLength)
30 {
31 }
32
33 // Create CAN message from long data in Network Order.
34 inline Message(uint32_t inIdentifier, uint8_t inLength, const uint64_t &inData, bool extended=false) :
35 identifier(inIdentifier), length(std::min(inLength, uint8_t(8)))
36 {
37 flags.extended = extended;
38 const uint8_t *inDataB = reinterpret_cast<const uint8_t *>(&inData);
39 for (uint8_t ii = 0; ii < length; ++ii)
40 data[ii] = inDataB[length - ii - 1];
41 }
42
43 inline uint32_t
44 getIdentifier() const
45 {
46 return identifier;
47 }
48
49 inline void
50 setIdentifier(uint32_t id)
51 {
52 identifier = id;
53 }
54
55 inline void
56 setExtended(bool extended = true)
57 {
58 flags.extended = (extended) ? 1 : 0;
59 }
60
61 inline bool
62 isExtended() const
63 {
64 return (flags.extended != 0);
65 }
66
67 inline void
68 setRemoteTransmitRequest(bool rtr = true)
69 {
70 flags.rtr = (rtr) ? 1 : 0;
71 }
72
73 inline bool
74 isRemoteTransmitRequest() const
75 {
76 return (flags.rtr != 0);
77 }
78
79 inline uint8_t
80 getLength() const
81 {
82 return length;
83 }
84
85 inline void
86 setLength(uint8_t len)
87 {
88 length = len;
89 }
90
91public:
92 uint32_t identifier;
93 uint8_t modm_aligned(4) data[8];
94 struct Flags
95 {
96 Flags() :
97 rtr(0), extended(1)
98 {
99 }
100
101 bool rtr : 1;
102 bool extended : 1;
103 } flags;
104 uint8_t length;
105
106public:
107 inline bool
108 operator == (const modm::can::Message& rhs) const
109 {
110 return ((this->identifier == rhs.identifier) and
111 (this->length == rhs.length) and
112 (this->flags.rtr == rhs.flags.rtr) and
113 (this->flags.extended == rhs.flags.extended) and
114 std::equal(data, data + length, rhs.data));
115 }
116
117 inline bool
118 operator < (const modm::can::Message& rhs) const
119 {
120 return (this->identifier << (this->flags.extended ? 0 : 18))
121 < (rhs.identifier << (rhs.flags.extended ? 0 : 18));
122 }
123};

134inline modm::IOStream&
135operator << (modm::IOStream& s, const modm::can::Message& m)
136{
137 s.printf("id = %04" PRIx32 ", len = ", m.identifier);
138 s << m.length;
139 s.printf(", flags = %c%c, data = ",
140 m.flags.rtr ? 'R' : 'r',
141 m.flags.extended ? 'E' : 'e');
142 if (not m.isRemoteTransmitRequest()) {
143 for (uint_fast8_t ii = 0; ii < m.length; ++ii) {
144 s.printf("%02x ", m.data[ii]);
145 }
146 }
147 return s;
148}
