159 #define MODM_IMPORT_BINARY(name, file, section) \
160 asm \
161 ( \
162 ".section " #section "\n" \
163 ".balign 4\n" \
164 ".global " #name "\n" \
165 #name ":\n" \
166 ".incbin \"" file "\"\n" \
167 ".global " #name "_laddr\n" \
168 ".set " #name "_laddr, . - " #name "\n" \
169 ".balign 4\n" \
170 ".section \".text\"\n" \
171 ); \
172 modm_extern_c uint8_t name ## _laddr[]; \
173 modm_extern_c const size_t name ## _length = (size_t) name ## _laddr; \
174 modm_extern_c uint8_t name[]
