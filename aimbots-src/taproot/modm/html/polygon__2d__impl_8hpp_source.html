56template <typename T>
57bool
58modm::Polygon2D<T>::intersects(const Polygon2D& other) const
59{
60 SizeType n = this->points.getSize();
61 SizeType m = other.points.getSize();
62
63 for (SizeType i = 0; i < n; ++i)
64 {
65 for (SizeType k = 0; k < m; ++k)
66 {
67 LineSegment2D<T> lineSegmentOwn(this->points[i], this->points[(i + 1) % n]);
68 LineSegment2D<T> lineSegmentOther(other.points[k], other.points[(k + 1) % m]);
69
70 if (lineSegmentOwn.intersects(lineSegmentOther)) {
71 return true;
72 }
73 }
74 }
75
76 return false;
77}

155template <typename T>
156bool
157modm::Polygon2D<T>::isInside(const modm::Polygon2D<T>::PointType& point)
158{
159 bool cw = true;
160 bool ccw = true;
161
162 SizeType n = this->points.getSize();
163 for (SizeType i = 0; i < n; ++i) {
164 int_fast8_t r = Vector<T, 2>::ccw(this->points[i], this->points[(i + 1) % n], point);
165
166 switch (r) {
167 case 0:
168 // point is one of the lines on the edge of the polygon.
169 return true;
170 break;
171
172 case 1:
173 cw = false;
174 break;
175
176 case -1:
177 ccw = false;
178 break;
179 }
180 }
181
182 return (cw || ccw);
183}
