115 template< class Scl, uint32_t baudrate = 100'000 >
116 static void
117 resetDevices()
118 {
119 static_assert(baudrate <= 500'000ul, "I2c::resetDevices() can only do max. 500kHz!");
120 constexpr auto delay = 500'000ul / baudrate;
121
122 for (uint_fast8_t ii = 0; ii < 9; ++ii) {
123 Scl::reset();
124 modm::delay_us(delay);
125 Scl::set();
126 modm::delay_us(delay);
127 }
128 }
