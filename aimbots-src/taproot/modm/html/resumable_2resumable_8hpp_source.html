106template< uint8_t Functions = 1>
107class Resumable
108{
109 static_assert(Functions > 0, "The number of resumable functions must be at least 1!");
110
111protected:
112 Resumable()
113 {
114 stopAllResumables();
115 }
116
117public:
119 inline void
120 stopAllResumables()
121 {
122 for (rf::State &state : rfStateArray)
123 {
124 state = rf::Stopped;
125 }
126 }
127
129 inline bool
130 stopResumable(uint8_t id)
131 {
132 if (id < Functions) {
133 rfStateArray[id] = rf::Stopped;
134 return true;
135 }
136 return false;
137 }
138
140 bool inline
141 isResumableRunning(uint8_t id) const
142 {
143 return (id < Functions and rfStateArray[id] != rf::Stopped);
144 }
145
147 bool inline
148 areAnyResumablesRunning() const
149 {
150 for (const rf::State state : rfStateArray)
151 {
152 if (state != rf::Stopped)
153 return true;
154 }
155 return false;
156 }
157
159 bool inline
160 areAnyResumablesRunning(std::initializer_list<uint8_t> ids) const
161 {
162 for (uint8_t id : ids)
163 {
164 if (isResumableRunning(id))
165 return true;
166 }
167 return false;
168 }
169
171 bool inline
172 areAllResumablesRunning(std::initializer_list<uint8_t> ids) const
173 {
174 for (uint8_t id : ids)
175 {
176 // = not isResumableRunning(id)
177 if (id >= Functions or rfStateArray[id] == rf::Stopped)
178 return false;
179 }
180 return true;
181 }
182
184 bool inline
185 joinResumables(std::initializer_list<uint8_t> ids) const
186 {
187 return not areAnyResumablesRunning(ids);
188 }
