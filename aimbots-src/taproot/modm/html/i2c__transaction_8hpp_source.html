241class I2cWriteReadTransaction : public I2cTransaction
242{
243public:
244 I2cWriteReadTransaction(uint8_t address) :
245 I2cTransaction(address), readSize(0), writeSize(0),
246 readBuffer(nullptr), writeBuffer(nullptr), isReading(false)
247 {
248 }
249
250 bool modm_always_inline
251 configurePing()
252 {
253 return configureWriteRead(nullptr, 0, nullptr, 0);
254 }
255
256 bool inline
257 configureWriteRead(const uint8_t *writeBuffer, std::size_t writeSize,
258 uint8_t *readBuffer, std::size_t readSize)
259 {
260 if (state != TransactionState::Busy)
261 {
262 this->readBuffer = readBuffer;
263 this->readSize = readSize;
264 this->writeBuffer = writeBuffer;
265 this->writeSize = writeSize;
266
267 /* `isReading` a bit tricky to understand, so here is an explanation:
268 *
269 * There are three possibilities right after an I2C start:
270 * 1. Stop communication (used for ping),
271 * 2. Start Writing something,
272 * 3. Start Reading something.
273 *
274 * Number 1 is easy, we can simply check if both readSize and writeSize are zero.
275 *
276 * 2 and 3 seem easy too, simply check if writeSize or readSize is non-zero respectively. No?
277 * NO! We want to keep the writeSize and readSize persistent, so that you do not need to
278 * reconfigure the adapter if nothing changed.
279 *
280 * So assume the following sequence with both writeSize and readSize non-zero:
281 * start - write - restart - read - ...
282 *
283 * During the first `starting()`, we check `writeSize != 0`, and request next operation `Write`.
284 * Then, during the next restart we will check `writeSize != 0`, but because it has not changed,
285 * we will never be able to enter the reading phase.
286 * We cannot check `readSize != 0` first either, because we encounter the same problem.
287 *
288 * Ok, so we use `isReading` here to signal, whether we want to write or read during `starting()`.
289 */
290 isReading = (writeSize == 0);
291 // however, there is another problem with `isReading`, see description in `writing()`
292
293 return true;
294 }
295 return false;
296 }
297

310 return configureWriteRead(buffer, size, nullptr, 0);

328protected:
329 Starting
330 starting() override
331 {
332 Starting starting(address, OperationAfterStart::Stop);
333 if (isReading) {
334 if (readSize) starting.next = OperationAfterStart::Read;
335 }
336 else {
337 if (writeSize) starting.next = OperationAfterStart::Write;
338 }
339
340 // This won't work here, see `writing()` for reason
341 // isReading = !isReading;
342
343 return starting;
344 }
345
346 Writing
347 writing() override
348 {
349 /* So the `isReading` adventure continues here.
350 * The code shown in `starting()` does work, but only for the specific sequence of this adapter:
351 * `start - write - restart - read - stop`
352 *
353 * The code starts to break once you have the sequence:
354 * `start - write - write - ...`
355 * because during the second write `isReading` is suddenly true and we either Stop or start Reading.
356 *
357 * While the I2C standard prohibits any continued readings, as in
358 * `start - read - read - ...`,
359 * the sequence:
360 * `start - read - restart - read - ...`
361 * would lead to similar problems.
362 *
363 * When would these sequences ever arise, you ask?
364 * Well, when this Adapter is subclassed and `writing()` is overwritten so that
365 * a few bytes can be prepended to the actual payload.
366 *
367 * This would be implemented as such (prepending 2 address bytes to a block of data taken from I2cEeprom):
368 *
369 * if (writeAddress) {
370 * writeAddress = false;
371 * return Writing(addressBuffer, 2, OperationAfterWrite::Write);
372 * }
373 * return I2cWriteReadTransaction::writing();
374 *
375 * This would call `starting()` twice, without modifying `isReading` until this method is called.
376 */
377 isReading = (readSize != 0);
378 return Writing(writeBuffer, writeSize,
379 isReading ? OperationAfterWrite::Restart : OperationAfterWrite::Stop);
380 }
381
382 Reading
383 reading() override
384 {
385 return Reading(readBuffer, readSize, OperationAfterRead::Stop);
386 }
387
388protected:
389 std::size_t readSize;
390 std::size_t writeSize;
391 uint8_t *readBuffer;
392 const uint8_t *writeBuffer;
393 bool isReading;
394};
