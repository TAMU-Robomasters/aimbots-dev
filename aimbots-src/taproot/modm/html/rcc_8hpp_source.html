236 static inline bool
237 enableClockOutput1(ClockOutput1Source src, uint8_t div)
238 {
239 uint32_t tmp = RCC->CFGR & ~(RCC_CFGR_MCO1 | RCC_CFGR_MCO1PRE);
240 if (div > 1) tmp |= (div + 2) << 24;
241 RCC->CFGR = tmp | uint32_t(src);
242 return true;
243 }
244
245 static inline bool
246 enableClockOutput2(ClockOutput2Source src, uint8_t div)
247 {
248 uint32_t tmp = RCC->CFGR & ~(RCC_CFGR_MCO2 | RCC_CFGR_MCO2PRE);
249 if (div > 1) tmp |= (div + 2) << 27;
250 RCC->CFGR = tmp | uint32_t(src);
251 return true;
252 }
