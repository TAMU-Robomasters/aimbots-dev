96 // External Interrupts
97 // Warning: This will disable any previously enabled interrupt which is
98 // routed to the same interupt line, e.g. PA3 will disable PB3.
99 // This is a hardware limitation by the STM32 EXTI.
100 inline static void enableExternalInterrupt()
101 {
102 // PA[x], x = 0 .. 3 maps to EXTICR[0]
103 // PA[x], x = 4 .. 7 maps to EXTICR[1]
104 // PA[x], x = 8 .. 11 maps to EXTICR[2]
105 // PA[x], x = 12 .. 15 maps to EXTICR[3]
106 // => bit3 and bit2 (mask 0x0c) specify the register
107 // => bit1 and bit0 (mask 0x03) specify the bit position
108 constexpr uint8_t index = (pin & 0b1100) >> 2;
109 constexpr uint8_t bit_pos = (pin & 0b0011) << 2;
110 constexpr uint16_t syscfg_mask = (0b1111) << bit_pos;
111 constexpr uint16_t syscfg_value = (port_nr & (0b1111)) << bit_pos;
112 SYSCFG->EXTICR[index] = (SYSCFG->EXTICR[index] & ~syscfg_mask) | syscfg_value;
113 EXTI->IMR |= mask;
114 }
115 inline static void disableExternalInterrupt() { EXTI->IMR &= ~mask; }
116 inline static void enableExternalInterruptVector(const uint32_t priority)
117 {
118 NVIC_SetPriority(ExternalInterruptIRQ, priority);
119 NVIC_EnableIRQ(ExternalInterruptIRQ);
120 }
121 inline static void disableExternalInterruptVector() { NVIC_DisableIRQ(ExternalInterruptIRQ); }
