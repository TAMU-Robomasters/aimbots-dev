30template< class Clock, class Duration >
31template< typename Rep, typename Period >
32void
33modm::GenericTimeout<Clock, Duration>::restart(std::chrono::duration<Rep, Period> interval)
34{
35 if (0 <= interval.count())
36 {
37 const auto cast_interval = std::chrono::duration_cast<duration>(interval);
38 modm_assert_continue_fail_debug(interval.count() <= duration::max().count(), "tmr.size",
39 "Timer interval must be smaller than the maximal duration!", interval.count());
40
41 _start = now();
42 _interval = cast_interval;
43 _state = ARMED;
44 }
45 else {
46 modm_assert_continue_fail_debug(false, "tmr.neg",
47 "Timer interval must be larger than zero!", interval.count());
48 stop();
49 }
50}

61template< class Clock, class Duration >
62bool
63modm::GenericTimeout<Clock, Duration>::execute()
64{
65 if (_state & (EXECUTED | STOPPED))
66 return false;
67
68 if ((_state == EXPIRED) or checkExpiration())
69 {
70 _state = (EXPIRED | EXECUTED);
71 return true;
72 }
73
74 // can only be fired once!
75 return false;
76}

78template< class Clock, class Duration >
79typename modm::GenericTimeout<Clock, Duration>::wide_signed_duration
80modm::GenericTimeout<Clock, Duration>::remaining() const
81{
82 if (_state == STOPPED)
83 return wide_signed_duration{0};
84 return std::chrono::duration_cast<wide_signed_duration>(_interval) +
85 std::chrono::time_point_cast<wide_signed_duration>(_start) -
86 std::chrono::time_point_cast<wide_signed_duration>(now());
87}
