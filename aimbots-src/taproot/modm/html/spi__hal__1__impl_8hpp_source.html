34void inline
35modm::platform::SpiHal1::initialize(Prescaler prescaler,
36 MasterSelection masterSelection, DataMode dataMode,
37 DataOrder dataOrder, DataSize dataSize)
38{
39 enable();
40 // disable peripheral
41 SPI1->CR1 &= ~SPI_CR1_SPE;
42 // set parameters
43 SPI1->CR1 = static_cast<uint32_t>(dataMode)
44 | static_cast<uint32_t>(dataOrder)
45 | static_cast<uint32_t>(masterSelection)
46 | static_cast<uint32_t>(prescaler)
47 | static_cast<uint32_t>(dataSize);
48 if(masterSelection == MasterSelection::Master) {
49 SPI1->CR2 |= SPI_CR2_SSOE; // for master mode
50 }
51 // reenable peripheral
52 SPI1->CR1 |= SPI_CR1_SPE;
53}
