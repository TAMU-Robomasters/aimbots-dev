54template< class SystemClock, modm::baudrate_t baudrate, modm::percent_t tolerance >
55void
56UsartHal3::initialize(Parity parity, WordLength length)
57{
58 enable();
59 disableOperation();
60
61 constexpr uint32_t scalar = (baudrate * 16 > SystemClock::Usart3) ? 8 : 16;
62 constexpr uint32_t max = ((scalar == 16) ? (1ul << 16) : (1ul << 15)) - 1ul;
63 constexpr auto result = Prescaler::from_range(SystemClock::Usart3, baudrate, 1, max);
64 modm::PeripheralDriver::assertBaudrateInTolerance< result.frequency, baudrate, tolerance >();
65
66 uint32_t cr1 = USART3->CR1;
67 // set baudrate and oversampling
68 if constexpr (scalar == 16) {
69 // When OVER8 = 0:, BRR[3:0] = USARTDIV[3:0].
70 USART3->BRR = result.prescaler;
71 cr1 &= ~USART_CR1_OVER8;
72 } else {
73 // When OVER8 = 1: BRR[15:4] = USARTDIV[15:4]
74 // BRR[2:0] = USARTDIV[3:0] shifted 1 bit to the right. BRR[3] must be kept cleared.
75 USART3->BRR = (result.prescaler & ~0b1111) | ((result.prescaler & 0b1111) >> 1);
76 cr1 |= USART_CR1_OVER8;
77 }
78 // Set parity
79 cr1 &= ~(USART_CR1_PCE | USART_CR1_PS);
80 cr1 |= static_cast<uint32_t>(parity);
81
82 // Set word length
83#ifdef USART_CR1_M1
84 cr1 &= ~(USART_CR1_M0 | USART_CR1_M1);
85#else
86 cr1 &= ~USART_CR1_M;
87#endif
88 cr1 |= static_cast<uint32_t>(length);
89
90 USART3->CR1 = cr1;
91}

93void
94UsartHal3::setSpiClock(SpiClock clk, LastBitClockPulse pulse)
95{
96 uint32_t cr2 = USART3->CR2;
97 cr2 &= ~(USART_CR2_LBCL | USART_CR2_CLKEN);
98 cr2 |= static_cast<uint32_t>(clk) | static_cast<uint32_t>(pulse);
99 USART3->CR2 = cr2;
100}

199void
200UsartHal3::acknowledgeInterruptFlags(InterruptFlag_t flags)
201{
202 /* Interrupts must be cleared manually by accessing SR and DR.
203 * Overrun Interrupt, Noise flag detected, Framing Error, Parity Error
204 * p779: "It is cleared by a software sequence (an read to the
205 * USART_SR register followed by a read to the USART_DR register"
206 */
207 if (flags.value & 0xful) {
208 uint32_t tmp;
209 tmp = USART3->SR;
210 tmp = USART3->DR;
211 (void) tmp;
212 }
213 (void) flags; // avoid compiler warning
214}
