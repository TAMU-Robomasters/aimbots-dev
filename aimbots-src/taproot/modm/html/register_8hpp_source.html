563template < typename Parent, typename Enum, typename Parent::UnderlyingType Mask, typename Parent::UnderlyingType Position = 0 >
564struct Configuration : public ::modm::FlagsOperators<typename Parent::EnumType, typename Parent::UnderlyingType>
565{
566private:
567 typedef typename Parent::UnderlyingType UnderlyingType;
568 typedef UnderlyingType UType; // lazy
569 typedef typename Parent::EnumType EType; // lazy
570
571public:
573 explicit constexpr Configuration(UnderlyingType config) :
574 FlagsOperators<EType, UType>(config << Position) {}
575
576 explicit constexpr Configuration(Enum config) :
577 FlagsOperators<EType, UType>(UType(config) << Position) {}
578
580 constexpr Configuration(Configuration const &o) = default;
582 constexpr Configuration(Configuration &&o) = default;
583
585
587 static inline void set(Parent &parent, Enum config)
588 { parent.value = (parent.value & ~(Mask << Position)) | (UType(config) << Position); }
589
591 static inline void reset(Parent &parent)
592 { parent.value &= ~(Mask << Position); }
593
595 static constexpr Enum get(Parent const &parent)
596 { return Enum((parent.value & (Mask << Position)) >> Position); }

630template < typename Parent, typename Parent::UnderlyingType Width, typename Parent::UnderlyingType Position = 0 >
631struct Value : public ::modm::FlagsOperators<typename Parent::EnumType, typename Parent::UnderlyingType>
632{
633private:
634 typedef typename Parent::UnderlyingType UnderlyingType;
635 typedef UnderlyingType UType; // lazy
636 typedef typename Parent::EnumType EType; // lazy
637
638 static constexpr UType Mask = ((1 << Width) - 1) << Position;
639public:
641 explicit constexpr Value(UnderlyingType config) :
642 FlagsOperators<EType, UType>((config << Position) & Mask) {}
643
645 constexpr Value(Value const &o) = default;
647 constexpr Value(Value &&o) = default;
648
650
652 static inline void set(Parent &parent, UnderlyingType config)
653 { parent.value = (parent.value & ~Mask) | ((config << Position) & Mask); }
654
656 static inline void reset(Parent &parent)
657 { parent.value &= ~Mask; }
658
660 static constexpr UType get(Parent const &parent)
661 { return (parent.value & Mask) >> Position; }
662
663
665 static constexpr Parent mask()
666 { return Parent(Mask); }

669 constexpr operator Parent() const
