33namespace aligned_storage_impl

35template<size_t Cap>
36union aligned_storage_helper
37{
38 struct double1 { double a; };
39 struct double4 { double a[4]; };
40 template<class T> using maybe = std::conditional_t<(Cap >= sizeof(T)), T, char>;
41 char real_data[Cap];
42 maybe<int> a;
43 maybe<long> b;
44 maybe<long long> c;
45 maybe<void*> d;
46 maybe<void(*)()> e;
47 maybe<double1> f;
48 maybe<double4> g;
49 maybe<long double> h;
50};
51} // namespace aligned_storage_impl
