155 template<class SystemClock>
156 static Value
157 setPeriod(uint32_t microseconds, bool autoApply = true)
158 {
159 // This will be inaccurate for non-smooth frequencies (last six digits
160 // unequal to zero)
161 uint32_t cycles = microseconds * (SystemClock::Timer10 / 1'000'000UL);
162 uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
163 Value overflow = cycles / prescaler;
164
165 overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
166
167 setPrescaler(prescaler);
168 setOverflow(overflow);
169
170 if (autoApply) {
171 // Generate Update Event to apply the new settings for ARR
172 TIM10->EGR |= TIM_EGR_UG;
173 }
174
175 return overflow;
176 }

240 /*
241 * Set Dead Time Value
242 *
243 * Different Resolution Depending on DeadTime[7:5]:
244 * 0xx => DeadTime[6:0] * T(DTS)
245 * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
246 * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
247 * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
248 */
249 static inline void
250 setDeadTime(uint8_t deadTime)
251 {
252 uint32_t flags = TIM10->BDTR;
253 flags &= ~TIM_BDTR_DTG;
254 flags |= deadTime;
255 TIM10->BDTR = flags;
256 }

258 /*
259 * Set Dead Time Value
260 *
261 * Different Resolution Depending on DeadTime[7:5]:
262 * 0xx => DeadTime[6:0] * T(DTS)
263 * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
264 * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
265 * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
266 */
267 static inline void
268 setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
269 {
270 uint8_t bitmask;
271 switch(resolution){
272 case DeadTimeResolution::From0With125nsStep:
273 bitmask = 0b01111111;
274 break;
275 case DeadTimeResolution::From16usWith250nsStep:
276 bitmask = 0b00111111;
277 break;
278 case DeadTimeResolution::From32usWith1usStep:
279 case DeadTimeResolution::From64usWith2usStep:
280 bitmask = 0b00011111;
281 break;
282 default:
283 bitmask = 0x00;
284 break;
285 }
286 uint32_t flags = TIM10->BDTR;
287 flags &= ~TIM_BDTR_DTG;
288 flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
289 TIM10->BDTR = flags;
290 }

308 static void
309 setInvertedPwm(uint32_t channel)
310 {
311 channel -= 1; // 1..2 -> 0..1
312
313 {
314 uint32_t flags = static_cast<uint32_t>(OutputCompareMode::Pwm2);
315
316 if (channel <= 1)
317 {
318 uint32_t offset = 8 * channel;
319
320 flags <<= offset;
321 flags |= TIM10->CCMR1 & ~(TIM_CCMR1_OC1M << offset);
322 TIM10->CCMR1 = flags;
323 }
324 else {
325 uint32_t offset = 8 * (channel - 2);
326
327 flags <<= offset;
328 flags |= TIM10->CCMR2 & ~(TIM_CCMR1_OC1M << offset);
329
330 TIM10->CCMR2 = flags;
331 }
332 }
333 }
