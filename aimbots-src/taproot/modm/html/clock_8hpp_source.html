114{
115 s << m.count();

117 if constexpr (std::is_same_v<R, std::micro>) s << "us";
118 if constexpr (std::is_same_v<R, std::milli>) s << "ms";
119 if constexpr (std::is_same_v<R, std::ratio<60>>) s << "min";
120 if constexpr (std::is_same_v<R, std::ratio<3600>>) s << 'h';
121 return s;
