157 template<class SystemClock>
158 static Value
159 setPeriod(uint32_t microseconds, bool autoApply = true)
160 {
161 // This will be inaccurate for non-smooth frequencies (last six digits
162 // unequal to zero)
163 uint32_t cycles = microseconds * (SystemClock::Timer4 / 1'000'000UL);
164 uint16_t prescaler = (cycles + 65'535) / 65'536; // always round up
165 Value overflow = cycles / prescaler;
166
167 overflow = overflow - 1; // e.g. 36'000 cycles are from 0 to 35'999
168
169 setPrescaler(prescaler);
170 setOverflow(overflow);
171
172 if (autoApply) {
173 // Generate Update Event to apply the new settings for ARR
174 TIM4->EGR |= TIM_EGR_UG;
175 }
176
177 return overflow;
178 }

242 /*
243 * Set Dead Time Value
244 *
245 * Different Resolution Depending on DeadTime[7:5]:
246 * 0xx => DeadTime[6:0] * T(DTS)
247 * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
248 * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
249 * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
250 */
251 static inline void
252 setDeadTime(uint8_t deadTime)
253 {
254 uint32_t flags = TIM4->BDTR;
255 flags &= ~TIM_BDTR_DTG;
256 flags |= deadTime;
257 TIM4->BDTR = flags;
258 }
259
260 /*
261 * Set Dead Time Value
262 *
263 * Different Resolution Depending on DeadTime[7:5]:
264 * 0xx => DeadTime[6:0] * T(DTS)
265 * 10x => (DeadTime[5:0] + 32) * 2 * T(DTS)
266 * 110 => (DeadTime[4:0] + 4) * 8 * T(DTS)
267 * 111 => (DeadTime[4:0] + 2) * 16 * T(DTS)
268 */
269 static inline void
270 setDeadTime(DeadTimeResolution resolution, uint8_t deadTime)
271 {
272 uint8_t bitmask;
273 switch(resolution){
274 case DeadTimeResolution::From0With125nsStep:
275 bitmask = 0b01111111;
276 break;
277 case DeadTimeResolution::From16usWith250nsStep:
278 bitmask = 0b00111111;
279 break;
280 case DeadTimeResolution::From32usWith1usStep:
281 case DeadTimeResolution::From64usWith2usStep:
282 bitmask = 0b00011111;
283 break;
284 default:
285 bitmask = 0x00;
286 break;
287 }
288 uint32_t flags = TIM4->BDTR;
289 flags &= ~TIM_BDTR_DTG;
290 flags |= (deadTime & bitmask) | static_cast<uint32_t>(resolution);
291 TIM4->BDTR = flags;
292 }

310 static void
311 setInvertedPwm(uint32_t channel)
312 {
313 channel -= 1; // 1..2 -> 0..1
314
315 {
316 uint32_t flags = static_cast<uint32_t>(OutputCompareMode::Pwm2);
317
318 if (channel <= 1)
319 {
320 uint32_t offset = 8 * channel;
321
322 flags <<= offset;
323 flags |= TIM4->CCMR1 & ~(TIM_CCMR1_OC1M << offset);
324 TIM4->CCMR1 = flags;
325 }
326 else {
327 uint32_t offset = 8 * (channel - 2);
328
329 flags <<= offset;
330 flags |= TIM4->CCMR2 & ~(TIM_CCMR1_OC1M << offset);
331
332 TIM4->CCMR2 = flags;
333 }
334 }
335 }
