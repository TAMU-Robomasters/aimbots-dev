37 modm_always_inline uint8_t
38 swap(uint8_t n)
39 {
40#ifdef MODM_CPU_AVR
41 if (__builtin_constant_p(n)) {
42 n = (n << 4) | (n >> 4);
43 }
44 else {
45 asm volatile ("swap %0" : "=r" (n) : "0" (n));
46 }
47 return n;
48#else
49 n = (n << 4) | (n >> 4);
50 return n;
51#endif
52 }

60 b = temp;

70 modm_always_inline uint16_t
71 swap(uint16_t n)
72 {
73#ifdef MODM_CPU_ARM
74 asm volatile(
75 "rev16 %0,%0" "\n\t"
76 : "=r" (n)
77 : "0" (n)
78 );
79 return n;
80#elif defined MODM_CPU_AVR
81 if (__builtin_constant_p(n)) {
82 n = (n << 8) | (n >> 8);
83 }
84 else {
85 asm volatile (
86 "eor %A0, %B0" "\n\t"
87 "eor %B0, %A0" "\n\t"
88 "eor %A0, %B0" "\n\t"
89 : "=r" (n)
90 : "0" (n)
91 );
92 }
93 return n;
94#else
95 n = (n << 8) | (n >> 8);
96 return n;
97#endif
98 }

107 modm_always_inline uint32_t
108 swap(uint32_t n)
109 {
110#if defined(MODM_CPU_ARM) && !defined(__aarch64__)
111 asm volatile(
112 "rev %0,%0" "\n\t"
113 : "=r" (n)
114 : "0" (n)
115 );
116 return n;
117#elif defined(MODM_CPU_ARM) && defined(__aarch64__)
118 asm volatile(
119 "rev32 %0,%0" "\n\t"
120 : "=r" (n)
121 : "0" (n)
122 );
123 return n;
124#else
125 n = (n << 24) | ((n << 8) & 0xff0000) | ((n >> 8) & 0xff00) | (n >> 24);
126 return n;
127#endif
128 }

148 inline uint8_t
149 bitReverse(uint8_t n)
150 {
151#if defined(MODM_CPU_ARM) && !defined(MODM_CPU_CORTEX_M0)
152 asm volatile(
153 "rbit %0,%0" "\n\t"
154 "rev %0,%0" "\n\t"
155 : "=r" (n)
156 : "0" (n)
157 );
158 return n;
159#else
160 n = ((uint8_t) (n >> 1) & 0x55) | ((uint8_t) (n << 1) & 0xaa);
161 n = ((uint8_t) (n >> 2) & 0x33) | ((uint8_t) (n << 2) & 0xcc);
162
163 return swap(n);
164#endif
165 }

170 inline uint16_t
171 bitReverse(uint16_t n)
172 {
173#if defined(MODM_CPU_ARM) && !defined(MODM_CPU_CORTEX_M0)
174 asm volatile(
175 "rbit %0,%0" "\n\t"
176 "rev16 %0,%0" "\n\t"
177 "rev %0,%0" "\n\t"
178 : "=r" (n)
179 : "0" (n)
180 );
181 return n;
182#else
183 n = ((n >> 1) & 0x5555) | ((n << 1) & 0xaaaa);
184 n = ((n >> 2) & 0x3333) | ((n << 2) & 0xcccc);
185 n = ((n >> 4) & 0x0f0f) | ((n << 4) & 0xf0f0);
186
187 return swap(n);
188#endif

194 inline uint32_t
195 bitReverse(uint32_t n)
196 {
197#if defined(MODM_CPU_ARM) && !defined(MODM_CPU_CORTEX_M0) && !defined(__aarch64__)
198 asm volatile(
199 "rbit %0,%0" "\n\t"
200 : "=r" (n)
201 : "0" (n)
202 );
203 return n;
204#elif defined(__aarch64__)
205 asm volatile(
206 "rbit %0,%0" "\n\t"
207 "rev32 %0,%0" "\n\t"
208 "rev %0,%0" "\n\t"
209 : "=r" (n)
210 : "0" (n)
211 );
212 return n;
213#else
214 n = ((n >> 1) & 0x55555555) | ((n << 1) & 0xaaaaaaaa);
215 n = ((n >> 2) & 0x33333333) | ((n << 2) & 0xcccccccc);
216 n = ((n >> 4) & 0x0f0f0f0f) | ((n << 4) & 0xf0f0f0f0);
217 n = ((n >> 8) & 0x00ff00ff) | ((n << 8) & 0xff00ff00);
218 n = ((n >> 16) & 0x0000ffff) | ((n << 16) & 0xffff0000);
219
220 return n;
221#endif
222 }

225 constexpr int8_t
226 leftmostBitHelper(uint32_t value)
227 {
228 return (value == 0)? 0 : (1 + leftmostBitHelper(value >> 1));
229 }
231
235 constexpr int8_t
236 leftmostBit(uint32_t value)
237 {
238 return leftmostBitHelper(value) - 1;
239 }
